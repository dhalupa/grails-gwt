021    package org.codehaus.groovy.grails.plugins.gwt;
022    
023    import java.io.IOException;
024    import java.util.Date;
025    import javax.servlet.*;
026    import javax.servlet.http.*;
027    
028    /**
029     * Servlet filter that ensures that the *.nocache.js files generated
030     * by GWT are not cached by the browser or a proxy. The source has been
031     * taken from <a href="http://seewah.blogspot.com/2009/02/gwt-tips-2-nocachejs-getting-cached-in.html">
032     * this blog</a>.
033     */
034    public class GwtCacheControlFilter implements Filter {
035        public void destroy() {}
036        public void init(FilterConfig config) throws ServletException {}
037    
038        public void doFilter(ServletRequest request, ServletResponse response, FilterChain filterChain)
039                throws IOException, ServletException {
040            HttpServletRequest httpRequest = (HttpServletRequest) request;
041            String requestURI = httpRequest.getRequestURI();
042            if (requestURI.contains(".nocache.")) {
043                long now = new Date().getTime();
044                HttpServletResponse httpResponse = (HttpServletResponse) response;
045                httpResponse.setDateHeader("Date", now);
046                httpResponse.setDateHeader("Expires", now - 86400000L);  // One day ago.
047                httpResponse.setHeader("Pragma", "no-cache");
048                httpResponse.setHeader("Cache-control", "no-cache, no-store, must-revalidate");
049            }
050            filterChain.doFilter(request, response);
051        }
052    }
